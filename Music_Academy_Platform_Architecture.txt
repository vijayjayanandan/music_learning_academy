Music Learning Platform - Complete Architecture Document
1. System Overview
1.1 Platform Vision
An open-source, feature-rich online music education platform that enables music teachers to deliver high-quality lessons globally, with self-hosting capabilities for individual instructors and scalability for institutional use.
1.2 Core Architecture Principles

Progressive Complexity: Start with monolith, evolve to microservices
Self-Hostable: Can run on a single VPS or scale to cloud infrastructure
No Vendor Lock-in: Use open-source alternatives wherever possible
Learning-Oriented: Each component demonstrates important technical concepts
Cost-Efficient: Minimize operational costs for individual teachers

-- Users table with multi-tenancy support
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255),
    role user_role NOT NULL DEFAULT 'student',
    profile JSONB DEFAULT '{}',
    preferences JSONB DEFAULT '{}',
    is_active BOOLEAN DEFAULT true,
    email_verified BOOLEAN DEFAULT false,
    school_id UUID, -- For multi-tenancy
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    deleted_at TIMESTAMPTZ, -- Soft delete
    
    -- Indexes
    INDEX idx_users_email ON users(email),
    INDEX idx_users_school ON users(school_id) WHERE school_id IS NOT NULL,
    INDEX idx_users_role ON users(role),
    INDEX idx_users_active ON users(is_active) WHERE is_active = true
);

-- Courses table
CREATE TABLE courses (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    teacher_id UUID REFERENCES users(id) ON DELETE SET NULL,
    title VARCHAR(500) NOT NULL,
    description TEXT,
    category VARCHAR(100),
    difficulty_level INTEGER CHECK (difficulty_level BETWEEN 1 AND 10),
    curriculum JSONB DEFAULT '{}', -- Flexible curriculum structure
    requirements JSONB DEFAULT '[]',
    outcomes JSONB DEFAULT '[]',
    price DECIMAL(10, 2),
    currency VARCHAR(3) DEFAULT 'USD',
    duration_weeks INTEGER,
    is_published BOOLEAN DEFAULT false,
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    
    -- Full text search
    search_vector tsvector GENERATED ALWAYS AS (
        setweight(to_tsvector('english', coalesce(title, '')), 'A') ||
        setweight(to_tsvector('english', coalesce(description, '')), 'B')
    ) STORED,
    
    INDEX idx_courses_teacher ON courses(teacher_id),
    INDEX idx_courses_published ON courses(is_published) WHERE is_published = true,
    INDEX idx_courses_search ON courses USING GIN(search_vector)
);

-- Class sessions with scheduling constraints
CREATE TABLE class_sessions (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    course_id UUID REFERENCES courses(id) ON DELETE CASCADE,
    teacher_id UUID REFERENCES users(id) ON DELETE CASCADE,
    student_id UUID REFERENCES users(id) ON DELETE CASCADE,
    scheduled_at TIMESTAMPTZ NOT NULL,
    duration_minutes INTEGER DEFAULT 60,
    status class_status DEFAULT 'scheduled',
    meeting_room_id VARCHAR(255), -- WebRTC room identifier
    recording_url TEXT,
    notes TEXT,
    teacher_feedback JSONB,
    student_rating INTEGER CHECK (student_rating BETWEEN 1 AND 5),
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    
    -- Prevent double-booking for teachers
    EXCLUDE USING gist (
        teacher_id WITH =,
        tstzrange(scheduled_at, scheduled_at + make_interval(mins => duration_minutes)) WITH &&
    ),
    
    -- Prevent double-booking for students  
    EXCLUDE USING gist (
        student_id WITH =,
        tstzrange(scheduled_at, scheduled_at + make_interval(mins => duration_minutes)) WITH &&
    ),
    
    INDEX idx_sessions_status ON class_sessions(status),
    INDEX idx_sessions_scheduled ON class_sessions(scheduled_at)
);

@Injectable()
export class PracticeAnalysisService {
  async analyzePitchAccuracy(audioBuffer: Float32Array): Promise<PitchAnalysis> {
    // Implement YIN algorithm
    const fundamental = this.detectPitch(audioBuffer);
    const targetNote = this.getTargetNote(fundamental);
    const cents = this.calculateCents(fundamental, targetNote);
    
    return {
      frequency: fundamental,
      note: targetNote,
      accuracy: Math.abs(cents) < 10 ? 'perfect' : 'needs_work',
      deviationCents: cents
    };
  }
  
  async analyzeRhythm(audioBuffer: Float32Array, targetBPM: number): Promise<RhythmAnalysis> {
    // Onset detection
    const onsets = this.detectOnsets(audioBuffer);
    const tempo = this.estimateTempo(onsets);
    const timing = this.analyzeTimingAccuracy(onsets, targetBPM);
    
    return {
      detectedBPM: tempo,
      targetBPM,
      timingAccuracy: timing.accuracy,
      rushingOrDragging: timing.tendency
    };
  }
}

typescript// payment.module.ts
interface PaymentModuleConfig {
  providers: {
    stripe: {
      publicKey: string;
      secretKey: string;
      webhookSecret: string;
    };
    // Add other providers as needed
  };
  
  escrow: {
    enabled: true,
    releaseAfterHours: 24,
    disputePeriodDays: 7
  };
  
  commission: {
    platformFee: 0.15, // 15%
    processingFee: 0.029, // 2.9% + 30c
    minPayout: 10.00
  };
}

@Injectable()
export class PaymentService {
  async createPaymentIntent(session: ClassSession): Promise<PaymentIntent> {
    // Calculate amounts
    const amount = session.course.price;
    const platformFee = amount * this.config.commission.platformFee;
    const teacherEarning = amount - platformFee;
    
    // Create Stripe payment intent
    const intent = await this.stripe.paymentIntents.create({
      amount: Math.round(amount * 100),
      currency: 'usd',
      metadata: {
        sessionId: session.id,
        studentId: session.student.id,
        teacherId: session.teacher.id
      }
    });
    
    // Store in database with escrow
    await this.createEscrowPayment({
      sessionId: session.id,
      amount,
      teacherEarning,
      releaseAt: addHours(new Date(), this.config.escrow.releaseAfterHours)
    });
    
    return intent;
  }
  
  async processEscrowReleases(): Promise<void> {
    // Cron job to release escrow payments
    const readyPayments = await this.getReleasableEscrowPayments();
    
    for (const payment of readyPayments) {
      await this.releaseToTeacher(payment);
    }
  }
}
